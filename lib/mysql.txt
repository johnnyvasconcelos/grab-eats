banco de dados chamado 'db_restaurante'
primeira tabela é 'dados' com os itens: nome_restaurante var(char25)
categoria varchar(15), abertura varchar(5), fechamento varchar(5), boas-vindas varchar(250)

a tabela 'pedidos' com os itens: id - autoincrement, nome - varchar(80), em_preparo - boolean (true ou false) ou numero comum para colocar 0 ou 1, preco - varchar(10) ou numero que possa ser separado por vírgula, mesa - numero normal, , para_levar (boolean), nome cliente, cpf

a tabela 'produtos' com os itens: id - autoincrement, foto - varchar(80), nome_produto - varchar(80), ingredientes - varchar(250) ou uma área para colocar lista de itens podendo ser até código html, preco - varchar(10) ou numero que possa ser separado por vírgula, categoria_produto varchar(10)

a tabela 'login' com usuario varchar(50), senha varchar(50)
